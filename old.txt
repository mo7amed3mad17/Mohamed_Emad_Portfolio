@app.route('/contact', methods=['POST', 'GET'])

    if request.method == 'POST':  # Check if the form is submitted
        try:
            name = request.form['name']  # Get name from form
            email = request.form['email']  # Get email from form
            subject = request.form['subject']  # Get subject from form
            message = request.form['message']  # Get message from form
        except KeyError as e:
            return f"Missing field: {str(e)}", 400  # Return an error if a field is missing
        
        # Create a new message using Flask-Mail
        msg = Message(subject, sender='mohamed123emad.17@gmail.com', recipients=['mohamed123emad.17@gmail.com'])  # Set recipients
        msg.body = f"Name: {name}\nEmail: {email}\n\nMessage:\n{message}"  # Set message body
        mail.send(msg)

    return render_template('Contact.html')

def send_email_via_gmail(credentials, msg):
    import base64
    from googleapiclient.discovery import build

    service = build('gmail', 'v1', credentials=credentials)
    raw_message = base64.urlsafe_b64encode(msg.as_bytes()).decode()
    message = {
        'raw': raw_message
    }
    service.users().messages().send(userId='me', body=message).execute()
